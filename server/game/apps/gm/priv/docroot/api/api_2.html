{

    "account_name", 
    "account_pwd"   
}

{

    "code",         
    "new_account_nam
    "account_id"    
}

2.4 获取权限("/account_sproto/get_pms")

请求格式
{

    "top_pms_id"    
}
回复格式
{

    "code",         
    "top_pms":[{
        "name",         
        "kv_list"       name, url, tab
    }],
    "pms":[{
        "name",         
        "kv_list"       name, url, tab
    }]
}
